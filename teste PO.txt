#include<stdio.h>

main ( ) {

	int n , k , F1 , F2 , F3 ;
	
	scanf ( " %d " , &n ) ;
	F1 = 0 ;
	F2 = 1 ;
	k = 1 ;

	while ( k <= n )
	{
		F3 = F1 + F2 ;
		F1 = F2 ;
		F2 = F3 ;
		k = k + 1 ;
	}	

	if ( F2 == F3 )
	{
		printf ( " olá " ) ;
	}

	else
	{
		k = 2 + 2 ;	
	}
	
	printf ( " %d %d " , n , F1 ) ;
}
